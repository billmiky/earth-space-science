emissionsDataFixed = {
  const url = "https://raw.githubusercontent.com/owid/co2-data/master/owid-co2-data.csv";
  
  // Load CSV
  const raw = await d3.csv(url);

  // Filter for "World" and the correct time range
  // We use d.country === 'World' which is often more reliable than ISO codes in this dataset
  const worldData = raw.filter(d => d.country === 'World' && +d.year >= 1850 && +d.year <= 2019);

  // Transform data (Wide to Long format for stacking)
  return worldData.flatMap(d => {
    // Force convert strings to numbers with "+"
    const coal = +d.coal_co2 || 0;
    const oil = +d.oil_co2 || 0;
    const gas = +d.gas_co2 || 0;
    const cement = +d.cement_co2 || 0;
    const flaring = +d.flaring_co2 || 0;
    const landUse = +d.land_use_change_co2 || 0;
    
    // Combine Cement and Flaring into "Others"
    const others = cement + flaring;

    // Return array of objects for this year (divide by 1000 for Gt)
    return [
      { year: +d.year, category: "Coal", emissions: coal / 1000 },
      { year: +d.year, category: "Oil", emissions: oil / 1000 },
      { year: +d.year, category: "Gas", emissions: gas / 1000 },
      { year: +d.year, category: "Others", emissions: others / 1000 },

    ];
  });
}

29.1 üîÑ Understanding Climate Feedbacks

29.1.2 Negative Feedback (Stabilizing)

29.3 üåç Why the Arctic Warms ~4x Faster

The Ice-Albedo Feedback explains Arctic amplification:

  1. Arctic starts bright: thick ice/snow with high albedo (~60%)
  2. Warming melts ice: exposes darker ocean/land (albedo ~6%)
  3. Energy jump: darker surfaces absorb ~10x more solar energy
  4. Faster local warming: extra absorbed heat accelerates melt
  5. Runaway loop: more melt ‚Üí lower albedo ‚Üí more absorption

This positive feedback is why the Arctic is the ‚Äúcanary in the coal mine.‚Äù It also explains why Arctic warming is much higher than the global average.

## üåç Why the Arctic Warms ~4x Faster

The **Ice-Albedo Feedback** explains Arctic amplification:

1. **Arctic starts bright**: thick ice/snow with high albedo (~60%)
2. **Warming melts ice**: exposes darker ocean/land (albedo ~6%)
3. **Energy jump**: darker surfaces absorb ~10x more solar energy
4. **Faster local warming**: extra absorbed heat accelerates melt
5. **Runaway loop**: more melt ‚Üí lower albedo ‚Üí more absorption

This positive feedback is why the Arctic is the "canary in the coal mine." It also explains why Arctic warming is much higher than the global average.
