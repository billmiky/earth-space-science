{
  // Simulated earthquake data along plate boundaries
  const quakes = [];
  // Ring of Fire
  for (let i = 0; i < 60; i++) {
    const angle = (i / 60) * Math.PI * 1.6 - Math.PI * 0.8;
    quakes.push({
      lon: -165 + 85 * Math.cos(angle) + (Math.random() - 0.5) * 6,
      lat: 8 + 38 * Math.sin(angle) + (Math.random() - 0.5) * 4,
      mag: 4.5 + Math.random() * 4,
      type: "Earthquake",
      region: "Ring of Fire"
    });
  }
  // Mid-Atlantic Ridge
  for (let i = 0; i < 25; i++) {
    quakes.push({
      lon: -32 + (Math.random() - 0.5) * 8,
      lat: -45 + i * 5 + (Math.random() - 0.5) * 3,
      mag: 3.5 + Math.random() * 2.5,
      type: "Earthquake",
      region: "Mid-Atlantic Ridge"
    });
  }
  // Alpine-Himalayan
  for (let i = 0; i < 25; i++) {
    quakes.push({
      lon: -5 + i * 5.5 + (Math.random() - 0.5) * 4,
      lat: 28 + Math.sin(i * 0.35) * 8 + (Math.random() - 0.5) * 5,
      mag: 4 + Math.random() * 3.5,
      type: "Earthquake",
      region: "Alpine-Himalayan Belt"
    });
  }
  // East African Rift
  for (let i = 0; i < 12; i++) {
    quakes.push({
      lon: 34 + (Math.random() - 0.5) * 4,
      lat: -8 + i * 3 + (Math.random() - 0.5) * 2,
      mag: 3.5 + Math.random() * 2,
      type: "Earthquake",
      region: "East African Rift"
    });
  }
  // Volcanoes
  const volcanoes = [
    {lon: -122.2, lat: 46.2, name: "Mt. St. Helens", type: "Volcano", region: "Ring of Fire", mag: 6},
    {lon: 105.4, lat: -6.1, name: "Krakatoa", type: "Volcano", region: "Ring of Fire", mag: 7},
    {lon: 14.4, lat: 40.8, name: "Vesuvius", type: "Volcano", region: "Alpine-Himalayan Belt", mag: 5},
    {lon: -155.3, lat: 19.4, name: "Kilauea", type: "Volcano", region: "Hotspot", mag: 5},
    {lon: 139.5, lat: 35.4, name: "Mt. Fuji", type: "Volcano", region: "Ring of Fire", mag: 5},
    {lon: -17.8, lat: 28.6, name: "Teide", type: "Volcano", region: "Hotspot", mag: 4},
    {lon: -64.7, lat: -18.2, name: "Ubinas", type: "Volcano", region: "Ring of Fire", mag: 4},
    {lon: 36.5, lat: -1.5, name: "Ol Doinyo Lengai", type: "Volcano", region: "East African Rift", mag: 4},
    {lon: -19.6, lat: 63.6, name: "Eyjafjallaj√∂kull", type: "Volcano", region: "Mid-Atlantic Ridge", mag: 5},
    {lon: 131.1, lat: 31.6, name: "Sakurajima", type: "Volcano", region: "Ring of Fire", mag: 4},
    {lon: 15.0, lat: 38.8, name: "Stromboli", type: "Volcano", region: "Alpine-Himalayan Belt", mag: 3.5},
  ];

  const allData = [...quakes, ...volcanoes];

  return Plot.plot({
    title: "üåç Global Seismic & Volcanic Activity",
    subtitle: "Why do these events cluster in narrow bands?",
    width: Math.min(800, window.innerWidth - 60),
    height: 420,
    projection: {type: "equirectangular"},
    color: {
      legend: true,
      domain: ["Earthquake", "Volcano"],
      range: ["#3182ce", "#e53e3e"]
    },
    marks: [
      Plot.geo({type: "Sphere"}, {stroke: "#cbd5e0", fill: "#f0f4f8"}),
      Plot.dot(allData, {
        x: "lon",
        y: "lat",
        r: d => d.type === "Volcano" ? 5 : d.mag * 0.6,
        fill: "type",
        fillOpacity: d => d.type === "Volcano" ? 0.9 : 0.45,
        stroke: d => d.type === "Volcano" ? "#c53030" : "#2b6cb0",
        strokeWidth: 0.5,
        symbol: d => d.type === "Volcano" ? "triangle" : "circle",
        tip: true,
        channels: {
          Region: "region",
          Magnitude: d => d.mag.toFixed(1),
          Name: d => d.name || ""
        }
      })
    ]
  });
}
